In order to compatible with C, only C identifier is allowed. To represent any other characters, we can use a underline followed by two hexagons, for example,

XX(_2B)/*+*/ \
XX(_3E)/*>*/ \
XX(_3C)/*<*/ \
XX(_3E_3D)/*>=*/ \
XX(_3C_3D)/*<=*/ \
XX(_3D_3D)/*==*/ \
XX(_25)/* % */ \
XX(_3C_3C)/*<<*/ \
#define EXEC_ONCE_TU_NAME "main"
#include <voba/value.h>
#include "m1/main.h"
#include "m2/main.h"

{
exec_once_init();
voba_value_t args[] = {0};

voba_apply(voba_symbol_value(s_hello_m2),voba_make_tuple(args));
return 0;
}
